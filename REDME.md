## TABLE NAME: cars (entity: car)

- id PK TINYINT NOTNULL
- maker VARCHAR(10) NOTNULL
- model VARCHAR(20) NOTNULL
- license_plate VARCHAR(10) NOTNULL
- production_year  YEAR NOTNULL
- color VARCHAR(10) NULL
- km SMALLINT NOTNULL
- fuel VARCHAR(10) NOTNULL
- condition VARCHAR(5) NOTNULL
- description TEXT NULL
- price SMALLINT NOTNULL
- created_at DATETIME NOTNULL
- updated_at DATETIME NOTNULL
